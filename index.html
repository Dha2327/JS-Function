Q7- Write a function that accepts parameter n and returns factorial of n


Ans-
function factorial(n){
  let answer = 1;
  if (n == 0 || n == 1){
    return answer;
  }
  else if(n > 1){
    for(var i = n; i >= 1; i--){
      answer = answer * i;
    }
    return answer;
  }
  else{
    return "number has to be positive."
  }  
}
let n = 5;
answer = factorial(n)
console.log( answer);

OUTPUT:25
